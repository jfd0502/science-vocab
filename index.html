    const canvas = document.getElementById('matrixCanvas');

    const ctx = canvas.getContext('2d');

    let columns = [];


    function resizeCanvas() { 

        canvas.width = window.innerWidth; canvas.height = window.innerHeight; 

        columns = Array(Math.floor(canvas.width / 25)).fill(0).map(() => Math.random() * -canvas.height); 

    }


    function updateMatrixWords() {

        let terms = [];

        for (let u in units) for (let l in units[u]) units[u][l].forEach(i => terms.push(i.t));

        if (terms.length > 0) allTerms = terms;

    }


    function drawMatrix() { 

        ctx.fillStyle = "rgba(10, 15, 28, 0.15)"; ctx.fillRect(0, 0, canvas.width, canvas.height); 

        ctx.fillStyle = "#3b82f6"; ctx.font = "bold 15px monospace"; 

        columns.forEach((y, i) => { 

            const word = allTerms[Math.floor(Math.random() * allTerms.length)]; 

            ctx.fillText(word, i * 28, y); 

            if (y > canvas.height && Math.random() > 0.975) columns[i] = 0; else columns[i] = y + 25; 

        }); 

    }

        fetchTeacherData(); resizeCanvas(); matrixInterval = setInterval(drawMatrix, 50);

    };

    async function fetchTeacherData(force = false) {

        try {

            const url = force ? `${SHEET_CSV_URL}&cachebust=${Date.now()}` : SHEET_CSV_URL;

            const response = await fetch(url); const csvText = await response.text();

            const rows = csvText.split(/\r?\n/).filter(r => r.trim().length > 0);

            const newUnits = {}; 

            

            rows.slice(1).forEach(row => {

                const cols = row.split(/,(?=(?:(?:[^"]*"){2})*[^"]*$)/).map(c => c.replace(/^"|"$/g, '').trim());

                if (cols.length >= 5) {

                    const term = cols[0]; const def = cols[1]; const img = cols[2];

                    const listName = cols[3]; const unitName = cols[4]; 

                    if (!newUnits[unitName]) newUnits[unitName] = {}; 

                    if (!newUnits[unitName][listName]) newUnits[unitName][listName] = []; 

                    newUnits[unitName][listName].push({t: term, d: def, img: img});

                }

            });

            units = newUnits; isDataLoaded = true; updateMatrixWords(); renderUnitSelection(); renderAdminUnitSelection();

        } catch (e) { console.error("Fetch Error:", e); }

    }

    async function deleteWord(u, l, idx) {

    async function handleAuth(type) {

        let payload = {}; const fullNameInput = document.getElementById('regFull').value;

        if (type === 'login') payload = { action: 'login', username: document.getElementById('userIn').value, password: document.getElementById('passIn').value };

        else { if(!fullNameInput) return alert("Please enter your full name."); payload = { action: 'register', fullname: fullNameInput, username: document.getElementById('regUser').value, password: document.getElementById('regPass').value }; }

        const btn = type === 'login' ? document.getElementById('loginSubmit') : document.getElementById('regSubmit');

        const oldText = btn.innerText; btn.innerText = "Verifying..."; btn.disabled = true;

        try { const response = await fetch(EXPORT_URL, { method: 'POST', body: JSON.stringify(payload) }); const result = await response.text(); if (result.startsWith("Success:")) { studentName = (type === 'register') ? fullNameInput : result.replace("Success: ", ""); localStorage.setItem('vocabSojournName', studentName); unlockApp(); } else { alert(result); } } catch (e) { alert("Network Error."); } finally { btn.innerText = oldText; btn.disabled = false; }

    }

    function resetQuizUI() { document.getElementById('quizDifficultyArea').classList.remove('hidden'); document.getElementById('quizPlayArea').classList.add('hidden'); document.getElementById('quizEnd').classList.add('hidden'); }

    function initQuiz(diff) {

        currentDifficulty = diff; const list = units[currentUnit][currentClass];

        let count = (diff === 'Easy') ? 5 : (diff === 'Medium') ? 10 : list.length;

        qPool = [...list].sort(() => 0.5 - Math.random()).slice(0, count);

        qScore = 0; qIdx = 0;

        document.getElementById('quizDifficultyArea').classList.add('hidden'); document.getElementById('quizPlayArea').classList.remove('hidden');

        document.getElementById('quizStats').innerText = `Accuracy: 0%`; renderQuiz();

    }

    function renderQuiz() { 

        if (qIdx >= qPool.length) { endQuiz(); return; } 

        const list = units[currentUnit][currentClass]; const correct = qPool[qIdx]; 

        const options = [correct, ...list.filter(i => i.t !== correct.t).sort(() => 0.5 - Math.random()).slice(0, 3)].sort(() => 0.5 - Math.random()); 

        document.getElementById('quizQuestion').innerText = `Def of: ${correct.t}?`; 

        document.getElementById('quizOptions').innerHTML = options.map(o => `<div class="nav-btn" style="margin-bottom: 10px; display: block; text-align: left; padding: 15px;" onclick="checkQuiz(this, ${o.t === correct.t})">${o.d}</div>`).join(''); 

        document.getElementById('nextQuizBtn').classList.add('hidden'); 

    }

    function checkQuiz(el, isCorrect) { 

        if (!document.getElementById('nextQuizBtn').classList.contains('hidden')) return; 

        if (isCorrect) { qScore++; el.style.background = 'var(--accent)'; } 

        else { el.style.background = 'var(--danger)'; } 

        qIdx++; document.getElementById('quizStats').innerText = `Accuracy: ${Math.round((qScore/qIdx)*100)}%`; 

        document.getElementById('nextQuizBtn').classList.remove('hidden'); 

    }

    function endQuiz() { 

        document.getElementById('quizPlayArea').classList.add('hidden'); 

        const pct = Math.round((qScore/qPool.length)*100) + "%"; 

        document.getElementById('quizEnd').innerHTML = `<h3>Result: ${pct} (${currentDifficulty})</h3><button class="main-btn" onclick="resetQuizUI()">Restart</button>`; 

        document.getElementById('quizEnd').classList.remove('hidden'); 

        addHistory(`Quiz (${currentDifficulty})`, pct); syncSessionToSheet(`Quiz (${currentDifficulty})`, pct); 

    }


    function resetMatchUI() { document.getElementById('matchDifficultyArea').classList.remove('hidden'); document.getElementById('matchPlayArea').classList.add('hidden'); document.getElementById('matchEnd').classList.add('hidden'); }

    function initMatch(diff) {

        currentDifficulty = diff; const list = units[currentUnit][currentClass];

        let count = (diff === 'Easy') ? 5 : (diff === 'Medium') ? 10 : list.length;

        const sub = [...list].sort(() => 0.5 - Math.random()).slice(0, count);

        mGoal = sub.length; mCorrect = 0; mTotal = 0;

        document.getElementById('matchDifficultyArea').classList.add('hidden'); document.getElementById('matchPlayArea').classList.remove('hidden');

        document.getElementById('matchStats').innerText = "Accuracy: 0%";

        let words = sub.map((x, i) => ({t: x.t, id: i, type: 'w'})).sort(() => 0.5 - Math.random()); 

        let defs = sub.map((x, i) => ({t: x.d, id: i, type: 'd'})).sort(() => 0.5 - Math.random()); 

        document.getElementById('matchContainer').innerHTML = `<div style="display:grid; grid-template-columns: 1fr 1fr; gap: 15px;"><div>${words.map(i => `<div class="nav-btn" draggable="true" ondragstart="drag(event)" data-id="${i.id}" data-type="w" style="margin-bottom:10px; padding:10px; font-size:0.8rem; border-style: dashed;">${i.t}</div>`).join('')}</div><div>${defs.map(i => `<div class="nav-btn" ondragover="event.preventDefault()" ondrop="drop(event)" data-id="${i.id}" data-type="d" style="margin-bottom:10px; padding:10px; font-size:0.8rem; border-style: dashed;">${i.t}</div>`).join('')}</div></div>`;

    }

    function drag(e) { draggedItem = e.target; }

    function drop(e) { 

        const target = e.target.closest('.nav-btn'); if (!target || !draggedItem) return; 

        mTotal++; 

        if (draggedItem.dataset.id === target.dataset.id) { 

            mCorrect++; target.style.background = 'var(--accent)'; 

            draggedItem.style.visibility = 'hidden'; target.draggable = false; 

        } else { 

            target.style.background = 'var(--danger)'; setTimeout(() => target.style.background = 'var(--glass)', 500); 

        } 

        const acc = Math.round((mCorrect / mTotal) * 100) + "%"; 

        document.getElementById('matchStats').innerText = `Accuracy: ${acc}`; 

        if (mCorrect === mGoal) { 

            document.getElementById('matchPlayArea').classList.add('hidden'); 

            document.getElementById('matchEnd').innerHTML = `<h3>Result: ${acc} (${currentDifficulty})</h3><button class="main-btn" onclick="resetMatchUI()">Restart</button>`; 

            document.getElementById('matchEnd').classList.remove('hidden'); 

            addHistory(`Match (${currentDifficulty})`, acc); syncSessionToSheet(`Match (${currentDifficulty})`, acc); 

        } 

    }

    async function fetchResultsAnalytics() { 

        try { 

            const response = await fetch(EXPORT_URL + "?action=getAnalytics&cb=" + Date.now()); 

            const data = await response.json(); 

            rawAnalytics = data.map(row => { 

                let u = 'N/A', l = 'N/A', name = 'Anonymous', m = 'N/A', s = '0%', t = 'No Date';

                if (Array.isArray(row)) { t = row[0] || t; name = row[1] || name; u = row[2] || u; l = row[3] || l; m = row[4] || m; s = row[5] || s; } 

                if (t !== 'No Date') { const dateObj = new Date(t); if (!isNaN(dateObj)) t = dateObj.toLocaleString([], { month: 'short', day: 'numeric', hour: '2-digit', minute: '2-digit' }); }

                if (!isNaN(s) && s !== '' && !String(s).includes('%')) s = Math.round(parseFloat(s) * 100) + "%";

                return { student: name, unit: u, list: l, mode: m, score: s, timestamp: t }; 

            }); 

            renderStudentList(); 

        } catch (e) { document.getElementById('studentListGrid').innerText = "Unable to load analytics."; } 

    }


    function renderStudentList() { 

        const grid = document.getElementById('studentListGrid'); 

        document.getElementById('individualStudentView').classList.add('hidden'); 

        grid.classList.remove('hidden'); 

        let studentSet = [...new Set(rawAnalytics.map(r => r.student))];

        if (studentSet.length > 1 && studentSet.includes('Anonymous')) studentSet = studentSet.filter(s => s !== 'Anonymous');

        if (studentSet.length === 0) { grid.innerHTML = '<div class="loading-text">No records found.</div>'; return; }

        grid.innerHTML = studentSet.map(s => `<button class="nav-btn" onclick="viewStudentDetail('${s}')">${s}</button>`).join(''); 

    }

    }
